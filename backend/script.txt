#!/bin/bash

echo "========================= Creando discos ========================="

MKDISK -size=10 -unit=M -fit=FF -path=/home/diegodubon/disk1.mia
MKDISK -size=10 -unit=M -fit=WF -path=/home/diegodubon/disk2.mia

FDISK -size=4 -unit=M -path=/home/diegodubon/disk1.mia -name=Part3 -type=P -fit=BF

FDISK -size=6 -unit=M -path=/home/diegodubon/disk3.mia -name=Part1 -type=P -fit=WF
FDISK -size=3 -unit=M -path=/home/diegodubon/disk3.mia -name=Part2 -type=P -fit=FF

MOUNT -path=/home/diegodubon/disk2.mia -name=Part2
MOUNT -path=/home/diegodubon/disk3.mia -name=Part1
MOUNT -path=/home/diegodubon/disk3.mia -name=Part2

echo "========================= Listando particiones montadas ========================="
